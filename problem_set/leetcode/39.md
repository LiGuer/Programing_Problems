# 39 Combination Sum
## Problem
    (input) $\boldsymbol a \in \mathbb Z_+^n, b \in \mathbb Z_+ \qu; a_i â‰¤ b$
    $$
      \boldsymbol x^T \boldsymbol a = b
      \boldsymbol x \in \mathbb N^n
    $$

class Solution {
public:
  int dot(vector<int>& a, vector<int>& b) {
    int n = a.size(), ans = 0;

    for (int i = 0; i < n; i++) {
      ans += a[i] * b[i];
    }

    return ans;
  }

  vector<vector<int>> combinationSum(vector<int>& candidates, int target) {
    vector<vector<int>> ans;
    vector<int> x;

    int n = candidates.size();
    for (int i = 0; i < n; i++) {
      x.push_back(0);
    }

    fun(0, x, candidates, target, ans);

    return ans;
  }

  void fun(int ind, vector<int>& x, vector<int>& a, int target, vector<vector<int>>& ans) {
    int n = a.size();

    if (ind == n - 1) {

      while (1) {
        int t = dot(x, a);

        for (int i = 0; i < x.size(); i++) {
          printf("%d ", x[i]);
        }printf(">%d \n", t);

        if (t > target) {
          x[ind] = 0;
          return;
        }

        else if (t == target) {
          getAns(x, a, ans);

          x[ind] = 0;
          return;
        }

        x[ind]++;
      }
    }

    while (1) {
      fun(ind + 1, x, a, target, ans);

      x[ind]++;

      int t = dot(x, a);
      if (t > target) {
        x[ind] = 0;
        return;
      }
    }

    x[ind] = 0;
    return;
  }

  void getAns(vector<int>& x, vector<int>& a, vector<vector<int>>& ans) {
    vector<int> t;
    int  n = a.size();

    for (int i = 0; i < n; i++) {
      for (int j = 0; j < x[i]; j++) {
        t.push_back(a[i]);
      }
    }
    ans.push_back(t);
  }
};
