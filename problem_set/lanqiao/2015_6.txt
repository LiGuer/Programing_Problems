#include<iostream>
#include<algorithm>
using namespace std;
int ans=0;
int judge(int a[])
{
	int t=0;
	for(int i=0;i<13;i++)t+=a[i];
	if(t==13)return 1;
	if(t<13)return 0;
	if(t>13)return -1;
}
void dfs(int a[],int k)
{
	if(judge(a)==1){ans++;for(int i=0;i<13;i++)cout<<a[i]<<' ';cout<<endl;return;}
	if(judge(a)==-1)return;
	if(k>=13)return;
	for(int i=0;i<=4;i++){
		a[k]=i;
		dfs(a,k+1);
		a[k]=0;
	}
}
int main()
{
	int a[15]={0};
	dfs(a,0);
	cout<<ans;
	return 0;
}
