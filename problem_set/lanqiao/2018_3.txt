31
--------------------------------------------------------------------
分解这些数的因数，看有多少个5，多少个2。
因为只有2x5，才能得到一个尾0。
5: 31。2: 99。
--------------------------------------------------------------------
#include <bits/stdc++.h>
using namespace std;
int main() {
	int factor[2]={0},k=100;
	while(k--){
        int t;
        scanf("%d",&t);
        while(t%5==0){
            factor[0]++;
            t/=5;
        }
        while(t%2==0){
            factor[1]++;
            t/=2;
        }
	}
	printf("5:%d 2:%d\n",factor[0],factor[1]);
	printf("%d\n",factor[0]<factor[1]?factor[0]:factor[1]);
}
